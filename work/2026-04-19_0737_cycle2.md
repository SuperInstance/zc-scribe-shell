Now let me examine each selected repo and score its documentation quality.

```bash
#!/bin/bash
# Examine and score documentation for selected repos

echo "# Documentation Quality Report" > work/doc_quality_report.md
echo "## Generated: $(date)" >> work/doc_quality_report.md
echo "" >> work/doc_quality_report.md

while IFS= read -r repo_path; do
    echo "## Repository: $repo_path" >> work/doc_quality_report.md
    
    # Check for README
    if [ -f "$repo_path/README.md" ]; then
        readme_size=$(wc -l < "$repo_path/README.md")
        readme_words=$(wc -w < "$repo_path/README.md")
        echo "- ✅ README.md exists ($readme_size lines, $readme_words words)" >> work/doc_quality_report.md
        
        # Quick quality check
        if grep -q "## " "$repo_path/README.md"; then
            echo "- ✅ README has sections" >> work/doc_quality_report.md
        else
            echo "- ⚠️ README lacks clear sections" >> work/doc_quality_report.md
        fi
        
        if grep -q "install\|setup\|usage" -i "$repo_path/README.md"; then
            echo "- ✅ README mentions installation/usage" >> work/doc_quality_report.md
        else
            echo "- ⚠️ README missing installation/usage info" >> work/doc_quality_report.md
        fi
    else
        echo "- ❌ No README.md found" >> work/doc_quality_report.md
    fi
    
    # Check for other documentation files
    doc_files=$(find "$repo_path" -maxdepth 1 -name "*.md" -o -name "*.txt" -o -name "*.rst" | grep -v README.md)
    if [ -n "$doc_files" ]; then
        echo "- 📚 Additional docs found:" >> work/doc_quality_report.md
        echo "$doc_files" | xargs -I {} echo "  - {}" >> work/doc_quality_report.md
    else
        echo "- 📝 No additional documentation files" >> work/doc_quality_report.md
    fi
    
    # Check for FLEET-RESEARCH.md
    if [ -f "$repo_path/FLEET-RESEARCH.md" ]; then
        echo "- 🚀 FLEET-RESEARCH.md exists" >> work/doc_quality_report.md
    else
        echo "- ❌ Missing FLEET-RESEARCH.md" >> work/doc_quality_report.md
    fi
    
    # Assign score (0-10)
    score=0
    if [ -f "$repo_path/README.md" ]; then
        score=$((score + 3))
        if [ $readme_words -gt 100 ]; then
            score=$((score + 2))
        fi
        if grep -q "## " "$repo_path/README.md"; then
            score=$((score + 1))
        fi
        if grep -q "install\|setup\|usage" -i "$repo_path/README.md"; then
            score=$((score + 1))
        fi
    fi
    
    if [ -f "$repo_path/FLEET-RESEARCH.md" ]; then
        score=$((score + 3))
    fi
    
    if [ -n "$doc_files" ]; then
        score=$((score + 1))
    fi
    
    echo "- **Documentation Score: $score/10**" >> work/doc_quality_report.md
    echo "" >> work/doc_quality_report.md
    
done < work/selected_repos.txt

echo "Report complete: work/doc_quality_report.md"
```
